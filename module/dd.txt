<script>

var e = document.querySelector('#totalPrice .total')

// Options for the observer (which mutations to observe)
const config = { attributes: true, childList: true, subtree: true };

// Callback function to execute when mutations are observed
const callback = function(mutationsList, observer) {
    for(const mutation of mutationsList) {
        if (mutation.type === 'childList') {
            reform_txt()
            console.log('A child node has been added or removed.');
        }
        else if (mutation.type === 'attributes') {
            console.log('The ' + mutation.attributeName + ' attribute was modified.');
        }
    }
};

// Create an observer instance linked to the callback function
const observer = new MutationObserver(callback);

// Start observing the target node for configured mutations
var e_MainOptionInBox = document.querySelector(".productOption optgroup > option")

if ( e_MainOptionInBox.textContent.includes("[예약주문]") ) {
    console.log("Set custom View");
    observer.observe(e, config);
}


    /**
      *		Custom Design of Sales Option
      */
    
function reform_txt() {
    if ( typeof document.querySelector("#option_box1_price .ec-front-product-item-price") === 'undefined' ) {
        return false;
    }

    let e_totCurPrice = document.querySelector("#option_box1_price .ec-front-product-item-price")
    let e_prdDisPrice = document.querySelector("#span_optimum_discount_price")
    let e_prdCurPrice = document.getElementById("span_product_price_text")

    // Calculate current number of product to buy
    cur_count = _get_price(e_totCurPrice.textContent) / _get_price(e_prdCurPrice.textContent)

    // Get total price before discount
    totPrice_text = e_totCurPrice.textContent.split('>')[0].split('<')[0]
    
    // Get total price after discount
    curPrice = _get_price(e_prdDisPrice.textContent) * cur_count
    e_totCurPrice.innerHTML = '<del style="font-size: 14px">' + totPrice_text + '</del><br><div style="color:#7f2562">' + curPrice.toLocaleString("en-US") + '원</div>'
}

function _get_price(str) {
    return parseInt(str.replace(/,*|\D/g, '').replace('원', ''))
}
